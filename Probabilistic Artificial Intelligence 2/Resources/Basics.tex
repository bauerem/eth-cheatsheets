\subsection*{Probability}
\term{Gauss:} $p(x)=\frac{1}{\sqrt{2\pi\sigma ^2}}\exp(-\frac{(x-\mu)^2}{2\sigma^2})$

$p(x)=\frac{1}{(2\pi)^{n/2}\sqrt{|\Sigma|}}\exp(-\frac{1}{2}(x-\mu)^T\Sigma^{-1}(x-\mu))$

\term{$\prod$Rule:} $P(A,B)=P(A|B)P(B)$

$P(X_{1:n})=P(X_1)P(X_2|X_1)P(X_3|X_{1:2})\cdots P(X_n|X_{1:n-1})$

\term{$\sum$Rule:} $P(X_{1:i-1},X_{i+1:n})=\sum_{x_i}P(X_{1:i-1},x_i,X_{i+1:n})$

$P(X_i)=\sum_{x_1,\dots,x_{i-1}}\sum_{x_{i+1},\dots,x_n}P(x_{1:i-1},X_i,x_{i+1:n})$
\subsection*{Posterior Inference}
\term{Prior:} $P(x)$
\term{Likelih.:} $P(x|y)$
\term{Posterior:} $P(y|x)$

\term{Bayes':} $P(X|Y)=\frac{P(X,Y)}{P(X)}=\frac{P(X)P(Y|X)}{\sum_x P(X=x)P(Y|X=x)}$
